    <script>
    document.addEventListener('DOMContentLoaded', function() {
        
        // --- 1. 時間割機能 ---
        const slotsInput = document.getElementById('id_available_slots');
        const slotBtns = document.querySelectorAll('.slot-btn');
        let selectedSlots = [];
        
        try {
            const rawValue = slotsInput.value;
            if (rawValue) {
                const validJson = rawValue.replace(/'/g, '"');
                const parsed = JSON.parse(validJson);
                
                // 【追加】データがちゃんと「配列」かチェックする
                if (Array.isArray(parsed)) {
                    selectedSlots = parsed;
                } else {
                    selectedSlots = [];
                }
            }
        } catch(e) {
            console.error("Parse error:", e);
            selectedSlots = [];
        }

        slotBtns.forEach(btn => {
            const val = btn.dataset.value;
            // 既存データを反映
            if (selectedSlots.includes(val)) btn.classList.add('active');

            // クリック時の挙動
            btn.addEventListener('click', function() {
                this.classList.toggle('active');
                const value = this.dataset.value;
                if (this.classList.contains('active')) {
                    if (!selectedSlots.includes(value)) selectedSlots.push(value);
                } else {
                    selectedSlots = selectedSlots.filter(item => item !== value);
                }
                slotsInput.value = JSON.stringify(selectedSlots);
            });
        });

        // --- 2. 参加条件のON/OFF制御機能 ---
        function setupToggle(checkboxName, optionsId) {
            const checkbox = document.querySelector(`input[name="${checkboxName}"]`);
            const optionsDiv = document.getElementById(optionsId);
            
            if (!checkbox || !optionsDiv) return;

            function toggle() {
                if (checkbox.checked) {
                    optionsDiv.style.display = 'block';
                } else {
                    optionsDiv.style.display = 'none';
                }
            }
            
            toggle(); // 初期表示
            checkbox.addEventListener('change', toggle); // 変更検知
        }

        setupToggle('enable_nationality', 'options_nationality');
        setupToggle('enable_gender', 'options_gender');
        setupToggle('enable_disease', 'options_disease');


        // --- 3. 人数入力機能 ---
        const selector = document.getElementById('id_participants_selector');
        const customInput = document.getElementById('id_custom_participants');
        
        function toggleCustomInput() {
            if (selector.value === 'custom') {
                customInput.style.display = 'inline-block';
                customInput.required = true;
            } else {
                customInput.style.display = 'none';
                customInput.required = false;
            }
        }
        toggleCustomInput();
        selector.addEventListener('change', toggleCustomInput);
    });
    </script>
